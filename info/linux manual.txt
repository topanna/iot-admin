4)	run as a service
sudo apt-get install -y systemd

sudo nano /etc/systemd/system/mtms.service

[Unit]
Description=MQTT to MySQL
After=multi-user.target
[Service]
Type=simple
Restart=always
ExecStart=/usr/bin/python3 /var/mtms/mqtttomysql.py
[Install]
WantedBy=multi-user.target

sudo systemctl daemon-reload

sudo systemctl enable mtms.service

sudo systemctl start mtms.service
